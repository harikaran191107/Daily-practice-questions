1)To print the area of rectangle:

#include<stdio.h>
int main(){
    int length,breadth,area;
    printf("enter the length:");
    scanf("%d",&length);
    printf("enter the breadth:");
    scanf("%d",&breadth);
    area=length*breadth;
    printf("area of the rectangle=%d units",area);
    return 0;
}

5)To print the distance between the two points:

#include<stdio.h>
#include<math.h>
int main(){
    int x1,x2,y1,y2,distance;
    printf("enter the point1:");
    scanf("%d%d",&x1,&y1);
    printf("enter the point2:");
    scanf("%d%d",&x2,&y2);
    distance=sqrt(((x2-x1)*(x2-x1))+((y2-y1)*(y2-y1)));
    printf("distance between the two points=%f",distance);
    return 0;
}

13)To print the surface area and volume of a cuboid:

#include<stdio.h>
int main(){
    int length,breadth,height,surfacearea,volume;
    printf("enter a length:");
    scanf("%d",&length);
    printf("enter a breadth:");
    scanf("%d",&breadth);
    printf("enter a height:");
    scanf("%d",&height);
    surfacearea=2*((length*breadth)+(breadth*height)+(height*length));
    volume=length*breadth*height;
    printf("surface area of the cuboid=%d\n units",surfacearea);
    printf("volume of a cuboid=%d units",volume);
    return 0;
}
